"Сколько всего было заказов?" ->
SELECT COUNT(DISTINCT order_id) AS total_orders
FROM incity;

SELECT COUNT(DISTINCT order_id) AS done_orders
FROM incity

SELECT COUNT(DISTINCT order_id) AS canceled_orders
FROM incity
WHERE status_order = 'cancel';

"Сколько заказов было удалено?" ->
SELECT COUNT(DISTINCT order_id) AS deleted_orders

WHERE status_order = 'delete';

"Сколько заказов было отклонено?" ->
SELECT COUNT(DISTINCT order_id) AS declined_orders

"Сколько всего было отклонений тендера?" ->
SELECT COUNT(*) AS declined_tenders

"Сколько заказов отменил клиент?" ->

"Сколько заказов отменил водитель?" ->
SELECT COUNT(DISTINCT order_id) AS driver_canceled_orders

"Сколько заказов отменили до принятия водителем?" ->
SELECT COUNT(DISTINCT order_id) AS canceled_before_accept
FROM incity

"Сколько заказов было без назначенного водителя?" ->
SELECT COUNT(DISTINCT order_id) AS orders_without_driver

"Сколько заказов было по каждому городу?" ->
SELECT city_id, COUNT(DISTINCT order_id) AS total_orders
FROM incity

ORDER BY total_orders DESC;

"В каком городе было больше всего заказов?" ->
SELECT city_id, COUNT(DISTINCT order_id) AS total_orders
FROM incity
GROUP BY city_id
ORDER BY total_orders DESC
LIMIT 1;

"Сколько отклонений было по каждому городу?" ->
SELECT city_id, COUNT(DISTINCT order_id) AS declined_orders

WHERE status_tender = 'decline'

ORDER BY declined_orders DESC;

"Сколько завершенных заказов было в конкретном городе?" ->
SELECT COUNT(DISTINCT order_id) AS done_orders

  AND status_order = 'done';

"Сколько заказов было по дням?" ->
SELECT DATE(order_timestamp) AS day, COUNT(DISTINCT order_id) AS total_orders
FROM incity
GROUP BY DATE(order_timestamp)
ORDER BY day;

"Сколько завершенных заказов было по дням?" ->
SELECT DATE(order_timestamp) AS day, COUNT(DISTINCT order_id) AS done_orders
FROM incity
WHERE status_order = 'done'
GROUP BY DATE(order_timestamp)
ORDER BY day;

"Сколько заказов было по неделям?" ->
SELECT DATE_TRUNC('week', order_timestamp) AS week, COUNT(DISTINCT order_id) AS total_orders
FROM incity
GROUP BY DATE_TRUNC('week', order_timestamp)
ORDER BY week;

"В какие часы заказов больше всего?" ->
SELECT EXTRACT(HOUR FROM order_timestamp) AS hour, COUNT(DISTINCT order_id) AS total_orders
FROM incity
GROUP BY EXTRACT(HOUR FROM order_timestamp)
ORDER BY total_orders DESC;

"В какие часы больше всего отмен?" ->
SELECT EXTRACT(HOUR FROM order_timestamp) AS hour, COUNT(DISTINCT order_id) AS canceled_orders

ORDER BY canceled_orders DESC;

"Какой общий оборот по завершенным заказам?" ->
SELECT SUM(price_order_local) AS total_revenue
FROM incity
WHERE status_order = 'done';

"Какой средний чек?" ->
SELECT AVG(price_order_local) AS avg_order_price
FROM incity
WHERE status_order = 'done';

"Какой средний чек по городам?" ->
SELECT city_id, AVG(price_order_local) AS avg_order_price
FROM incity
WHERE status_order = 'done'
GROUP BY city_id
ORDER BY avg_order_price DESC;

"Какая максимальная стоимость заказа?" ->
SELECT MAX(price_order_local) AS max_order_price
FROM incity;

"Какая минимальная стоимость заказа?" ->
SELECT MIN(price_order_local) AS min_order_price
FROM incity;

"Насколько итоговая цена отличается от стартовой в среднем?" ->
SELECT AVG(price_order_local - price_start_local) AS avg_price_diff
FROM incity
WHERE status_order = 'done';

"Насколько цена тендера отличается от итоговой в среднем?" ->
SELECT AVG(price_order_local - price_tender_local) AS avg_tender_to_final_diff
FROM incity
WHERE status_order = 'done'
  AND price_tender_local IS NOT NULL;

"Сколько в среднем проходит от создания заказа до принятия водителем?" ->
SELECT AVG(EXTRACT(EPOCH FROM (driveraccept_timestamp - order_timestamp)) / 60.0) AS avg_minutes_to_accept
FROM incity
WHERE driveraccept_timestamp IS NOT NULL;

"Сколько в среднем проходит от принятия до прибытия водителя?" ->
SELECT AVG(EXTRACT(EPOCH FROM (driverarrived_timestamp - driveraccept_timestamp)) / 60.0) AS avg_minutes_to_arrive
FROM incity
WHERE driveraccept_timestamp IS NOT NULL
  AND driverarrived_timestamp IS NOT NULL;

"Сколько в среднем длится поездка?" ->
SELECT AVG(duration_in_seconds) / 60.0 AS avg_ride_minutes
FROM incity
WHERE status_order = 'done';

"Какое среднее расстояние поездки?" ->
SELECT AVG(distance_in_meters) / 1000.0 AS avg_distance_km
FROM incity
WHERE status_order = 'done';

"Сколько в среднем проходит от старта поездки до завершения?" ->
SELECT AVG(EXTRACT(EPOCH FROM (driverdone_timestamp - driverstarttheride_timestamp)) / 60.0) AS avg_minutes_ride
FROM incity
WHERE driverstarttheride_timestamp IS NOT NULL
  AND driverdone_timestamp IS NOT NULL;

"Какой процент заказов был завершен?" ->
SELECT
  ROUND(
    100.0 * COUNT(DISTINCT CASE WHEN status_order = 'done' THEN order_id END)
    / COUNT(DISTINCT order_id),
    2
  ) AS done_rate_percent
FROM incity;

"Какой процент заказов был отменен?" ->
SELECT
  ROUND(
    100.0 * COUNT(DISTINCT CASE WHEN status_order = 'cancel' THEN order_id END)
    / COUNT(DISTINCT order_id),
    2
  ) AS cancel_rate_percent
FROM incity;

"Какой процент заказов имел хотя бы одно отклонение тендера?" ->
SELECT
  ROUND(
    100.0 * COUNT(DISTINCT CASE WHEN status_tender = 'decline' THEN order_id END)
    / COUNT(DISTINCT order_id),
    2
  ) AS decline_rate_percent
FROM incity;

"Какой процент заказов дошел до назначения водителя?" ->

    / COUNT(DISTINCT order_id),

  ) AS driver_assigned_rate_percent
FROM incity;
